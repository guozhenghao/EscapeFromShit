/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func mergeKLists(lists []*ListNode) *ListNode {
    if lists == nil || len(lists) == 0{
        return nil
    }
    return merge(lists, 0, len(lists)-1)
}

func merge(lists []*ListNode, startIndex, endIndex int) *ListNode{
    if startIndex == endIndex{
        return lists[startIndex]
    }
    midIndex := startIndex + (endIndex - startIndex) / 2
    leftList := merge(lists, startIndex, midIndex)
    rightList := merge(lists, midIndex+1, endIndex)
    return mergeTwoLists(leftList, rightList)
}

func mergeTwoLists(list1, list2 *ListNode) *ListNode{
    if list1 == nil{
        return list2
    }
    if list2 == nil{
        return list1
    }
    resultStart := &ListNode{}
    result := resultStart
    for list1 != nil && list2 != nil{
        if list1.Val < list2.Val{
            result.Next = list1
            list1 = list1.Next
        }else{
            result.Next = list2
            list2 = list2.Next
        }
        result = result.Next
    }
    if list1 != nil{
        result.Next = list1
    }
    if list2 != nil{
        result.Next = list2
    }
    return resultStart.Next
}

// 一个大佬的答案
func reverseBetween(head *ListNode, m int, n int) *ListNode {
	tmpHead := &ListNode{Val: 0, Next: head}
	pre := tmpHead

	// pre 定位到m之前的元素上
	for i := 0; i < m-1; i++ {
		pre = pre.Next
	}

	if pre.Next == nil {
		return head
	}

	// 部分翻转
	curr := pre.Next
	// 循环m-n
	//
	for i := 0; i < n-m; i++ {
		tmp := pre.Next            // 暂存pre后面的节点
		pre.Next = curr.Next       // 断开连接，连接到下一个节点上
		curr.Next = curr.Next.Next // 更新Next到下一个
		pre.Next.Next = tmp        // 将暂存节点放在 pre-Next的后面
	}
	return tmpHead.Next
}
